jquery.spinner

spinner on click

$('form input').clickSpinner();

clear spinner

$('.hide-spinner').click(function() {
  $('form input').hideSpinner();
  $('form').hideSpinner();
});

spinner on submit

$('form').submitSpinner();

spinner on specific element

$('form').submitSpinner({target: '.submit2'});

spinner off

$('form input').clickSpinner({
  onCreate: function(spinner) {
    setTimeout(function() {
      spinner.hide();
    }, 1000);
  }
});

spinner on field

$('form input').customSpinner(function(spinner) {
  $(this).focus(function() {
    spinner.show($(this));
  });
  $(this).blur(function() {
    spinner.hide();
  });
});

custom spinner object

$('form input').clickSpinner({
  create: function() {
    var element = document.createElement('span');
    element.innerText = 'Loading ...';
    this.created(element);
  },
  onCreate: function(spinner) {
    setTimeout(function() { spinner.hide(); }, 1000);
  }
});
